31inline void CanSvcEventConsume(VlcbNetWireEndpointHandle endpoint,
32 CanSvcEvent event) {
33 switch (event.type) {
34 case CAN_PKT_INGEST:
35 if (endpoint->meta.can.state == VLCB_NET_WIRE_CAN_STATE_INITIALIZED) {
36 if (endpoint->addr.can == event.data.ingest.pkt->srcAddr.can) {
37 // handle collisions
38 endpoint->meta.can.state = VLCB_NET_WIRE_CAN_STATE_UNINITIALIZED;
39 endpoint->addr.can = VLCB_CAN_ID_EMPTY;
40 } else if (event.data.ingest.pkt->meta.can.is_rtr) {
41 // handle response to enumeration of other nodes
42 endpoint->meta.can.state = VLCB_NET_WIRE_CAN_STATE_RESPONDING;
43 }
44 } else if (endpoint->meta.can.state ==
45 VLCB_NET_WIRE_CAN_STATE_ENUMERATING) {
46 // handle enumeration
47 ToggleUsedId(&endpoint->meta.can.occupiedIdCache,
48 event.data.ingest.pkt->srcAddr.can);
49 }
50 break;
51 }
52}
