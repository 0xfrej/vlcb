22static inline void HandleHeartbeat(VlcbModule *const self, const clock_t now) {
23 // send heartbeat if module is in normal mode, heartbeat is enabled and
24 // heartbeat timeout passed (or if it's the first heartbeat) heartbeatSequence
25 // increments, until it overflows, which is expected
26 if (self->sm.state == VLCB_MODULE_STATE_NORMAL &&
27 self->config.operationFlags & VLCB_MODULE_FLAGS_HEARTBEAT &&
28 (self->lastHeartbeat == 0 ||
29 vlcb_platform_time_DiffInMs(self->lastHeartbeat, now) >=
30 VLCB_MODULE_HEARTBEAT_MS)) {
31 VlcbNetPacketDatagram packet;
32 vlcb_net_pkt_dgram_module_Heartbeat_Serialize(
33 &packet, (VlcbNetDgramHeartbeat){
34 .nodeNumber = self->config.nodeNumber,
35 .sequence = self->heartbeatSequence,
36 .status = 0, // TODO: replace with diagnostic status
37 });
38
39 VlcbNetSocketDgramSendErr err =
40 vlcb_net_sock_dgram_Send(self->socket, &packet);
41 // we can ignore misseed heartbeats due to socket buffer being full
42 if (err != VLCB_NET_SOCK_DGRAM_SEND_ERR_OK) {
43 if (err != VLCB_NET_SOCK_DGRAM_SEND_ERR_BUF_FULL) {
44 VLCBLOG_ERROR(vlcb_net_sock_dgram_SendErrToStr(err));
45 }
46 return; // ensure that when socket errors out, we don't update heartbeat
47 // stats
48 }
49
50 self->lastHeartbeat = now;
51 self->heartbeatSequence++;
52 }
53}

247void vlcb_module_Poll(VlcbModule *const module, const clock_t now) {
248 assert(module != NULL && module->iface != NULL);
249
250 VlcbNetPacketDatagram packet;
251
252 while (1) {
253 const VlcbNetSocketDgramRecvErr err =
254 vlcb_net_sock_dgram_Recv(module->socket, &packet);
255 if (err != VLCB_NET_SOCK_DGRAM_RECV_ERR_OK) {
256 if (err == VLCB_NET_SOCK_DGRAM_RECV_ERR_WOULD_BLOCK) {
257 break; // nothing to receive
258 }
259 }
260 HandleMnsMessages(module, &packet, now);
261
262 if (module->sm.state == VLCB_MODULE_STATE_NORMAL) {
263 // TODO: handle services
264 }
265 }
266
267 HandleHeartbeat(module, now);
268
269 _INTERFACE_CALL(module->ui, Poll, now);
270}
